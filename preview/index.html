Use it with any stack

SAPL provides SDKs with idiomatic framework integration. One annotation or decorator to enforce policies — same authorization model everywhere.

@PreEnforce(action = "'read'", resource = "'patient'")
public Patient getPatient(@PathVariable String id) {
    return patientRepository.findById(id);
}
@pre_enforce(action="read", resource="patient")
async def get_patient(patient_id: str):
    return patient_service.find_by_id(patient_id)
@PreEnforce({ action: 'read', resource: 'patient' })
async getPatient(@Param('id') id: string) {
    return this.patientService.findById(id);
}
[PreEnforce(Action = "read", Resource = "patient")]
public async Task<Patient> GetPatient(string id)
{
    return await _patientService.FindByIdAsync(id);
}
@pre_enforce(action="read", resource="patient")
async def get_patient(request, patient_id):
    return await patient_service.find_by_id(patient_id)
